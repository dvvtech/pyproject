private void WaitConnection(Socket socket, SocketAsyncEventArgs context)
{
    try
    {
        context.AcceptSocket = null;
        if (socket.AcceptAsync(context))
            return; // Асинхронное завершение

        Task.Run(() => ProcessConnection(context));
    }
    catch (Exception ex)
    {
        _diagnosticsContext.LogError(ex, "WaitConnection failed");
    }
}
